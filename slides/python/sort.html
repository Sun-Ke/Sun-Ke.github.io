                                def bubble_sort(arr):
                                    n = len(arr)
                                    for i in range(n):                  # 外层循环控制遍历次数
                                        for j in range(n - 1 - i):      # 内层循环比较相邻元素
                                            if arr[j] > arr[j + 1]:     # 如果前一个元素比后一个大
                                                arr[j], arr[j + 1] = arr[j + 1], arr[j]  # 交换两者的位置
                                        print(f"第 {i + 1} 次排序结果: {arr}")  # 打印每次排序后的数组状态

                                array = [5, 3, 8, 1, 6, 2]
                                print("原始数组:", array)
                                bubble_sort(array)
                                print("排序后数组:", array)

                    def selection_sort(arr):
                        n = len(arr)
                        for i in range(n):                      # 外层循环控制已排序部分
                            min_index = i                       # 假设当前索引为最小值索引
                            for j in range(i + 1, n):           # 在未排序部分寻找最小值
                                if arr[j] < arr[min_index]:     # 找到更小的值
                                    min_index = j               # 更新最小值索引
                            arr[i], arr[min_index] = arr[min_index], arr[i]  # 交换当前元素和最小值
                            print(f"第 {i + 1} 次排序结果: {arr}")  # 打印每次排序后的数组状态
                
                    array = [5, 3, 8, 1, 6, 2]
                    print("原始数组:", array)
                    selection_sort(array)
                    print("排序后数组:", array)
